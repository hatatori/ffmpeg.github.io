recorte tempo

	ffmpeg -i [inputfile] -ss 00:41 -t 00:10 -async 1 [outputfile]

	mater qualidade

	para mp4 ultiliza o -crf

	ffmpeg -i [inputfile] -crf 10 [outputfile]

	compilação de qualidade o quanto maior mais maneiro
	para mp4 ultiliza o -avi

	ffmpeg -i [inputfile] -q 23 [outputfile]

	qualidade e tempo

	ffmpeg -i [inputfile] -ss 00:41 -t 00:10 -async 1 -crf 23 out3.mkv

	qualidade de audio

	ffmpeg -i [inputfile] -b:a 320k [outputfile]

	to target the video

	ffmpeg -i [inputfile] -b:1000k [outputfile]

	aumenta diminui volume

	ffmpeg -i [inputfile] -filter:a "volume=2" [outputfile]
	ffmpeg -i [inputfile] -filter:a "channelmap=0-0|0-1" [outputfile]

	cotar

	ffmpeg -i [inputfile] -filter:v "crop=w=640:h=480:x=100:y=100" [outputfile]

	tamanho scale

	ffmpeg -i [inputfile] -filter:v "scale=w=200:h=-1" [outputfile]	
	ffmpeg -i out3.mkv -filter:v "scale=w=200:h=-1" outout.mp4


	rotacionar

	1 = 90 graus
	ffmpeg -i out3.mkv -vf "transpose=1" rotate.mp4

	// pi em radiano
	ffmpeg -i [input] -filter:v "rotate=30*PI/180" [output]
